__host__ __device__ inline bool IsVEGA20(ChipID id)
{
  switch (id) {
    case ChipID::VEGA20_INSTINCT:
      return true;
    case ChipID::VEGA20_MI50:
      return true;
    case ChipID::VEGA20:
      return true;
    case ChipID::VEGA20_VEGAII:
      return true;
    case ChipID::VEGA20_VII:
      return true;
    default:
      return false;
  }
}
